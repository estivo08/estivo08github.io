    <script>

        // --- DOM Elements ---

        const copyBtn = document.getElementById('copy-code-btn');

        const codeBlock = document.getElementById('applescript-code');


        /**

         * Copies the AppleScript code to the clipboard.

         */

        function copyCodeToClipboard() {

            // Get the text content from the code block

            const textToCopy = codeBlock.textContent;

            

            // We use a temporary textarea to hold the text for copying

            // This is a robust way to copy text in all browsers, including iFrames

            const tempTextArea = document.createElement('textarea');

            tempTextArea.value = textToCopy;

            document.body.appendChild(tempTextArea);

            tempTextArea.select();

            tempTextArea.setSelectionRange(0, 99999); // For mobile devices


            try {

                document.execCommand('copy');

                copyBtn.textContent = 'Copied!';

                setTimeout(() => {

                    copyBtn.textContent = 'Copy Code';

                }, 2000);

            } catch (err) {

                console.error('Failed to copy text: ', err);

                copyBtn.textContent = 'Failed!';

            }

            

            // Clean up the temporary textarea

            document.body.removeChild(tempTextArea);

        }


        // --- Event Listeners ---

        copyBtn.addEventListener('click', copyCodeToClipboard);

    </script>
